### ✔ Array 객체
- 다양한 유형의 다수 데이터를 순차적으로 저장하고 관리하는 내장객체
- 객체 생성
  ```
  // <방법 1>
  var 객체참조변수 = new Array(데이터1, ~, 데이터n);

  // <방법 2> ***
  var 변수명 = new Array();
  변수명.push(데이터1);
  ~
  변수명.push(데이터n);

  // <방법 3>
  var 변수명 = new Array();
  변수명[0] = 데이터1;
  ~
  변수명[n-1] = 데이터n;

  // <방법 4> ***
  var 변수명 = [데이터1, ~, 데이터n];

  // <방법 5> ***
  var 변수명 = [];
  변수명.push(데이터1);
  ~
  변수명.push(데이터n);
  ```
